FAST iShowBMP2 function

*/

void iShowBMP2(int x, int y, struct pic *a, int ignr0 = -1, int ignr1=-1, int ignr2=-1){
	int xx,yy;
	xx = a->x,yy=a->y;
	int arr[4],temp;
	for(int i = 0; i<xx;i++){
		for(int j = 0; j<yy;j++){
			temp = a->pxl[i][j];
			arr[0] = temp / 1000000;
			temp%=1000000;
			arr[1] = temp / 1000;
			temp %= 1000;
			arr[2] = temp;
			if(arr[0]==ignr0 && arr[1]==ignr1 && arr[2]==ignr2) continue;
			iSetColor(arr[0],arr[1],arr[2]);
			iPoint(i+x,j+y);
		}
	}
}
